# 1) Register a user
curl -k -X POST https://localhost:7047/api/auth/register \
  -H "Content-Type: application/json" \
  -d '{"username":"admin","email":"admin@test.com","password":"Admin123!","name":"Admin","surname":"User","birthDate":"1995-01-01"}'

# 2) Login and get userId
curl -k -X POST https://localhost:7047/api/auth/login \
  -H "Content-Type: application/json" \
  -d '{"username":"admin","password":"Admin123!"}'
# note the returned "UserId" value (example: 1)

# 3) Add base lookup data
curl -k -X POST https://localhost:7047/api/drinkDb/glasses \
  -H "Content-Type: application/json" \
  -d '[{"name":"Highball glass"}]'

curl -k -X POST https://localhost:7047/api/drinkDb/ingredients \
  -H "Content-Type: application/json" \
  -d '[{"name":"Vodka"},{"name":"Orange juice"}]'

# 4) Read IDs
curl -k https://localhost:7047/api/drinkDb/glasses
curl -k https://localhost:7047/api/drinkDb/ingredients
# note glass id + ingredient ids

# 5) Create one drink (replace IDs and UserId)
curl -k -X POST https://localhost:7047/api/drinkDb/createdrink \
  -H "Content-Type: application/json" \
  -d '{"name":"Screwdriver","category":"Cocktail","glassId":1,"userId":1,"instructions":"Mix and serve.","ingredients":[{"ingredientId":1,"amount":"50 ml"},{"ingredientId":2,"amount":"100 ml"}]}'

# 6) Get drink id
curl -k https://localhost:7047/api/drinkDb

# 7) Create active selection (replace userId + drinkId)
curl -k -X POST https://localhost:7047/api/selection/add-selection \
  -H "Content-Type: application/json" \
  -d '{"userId":1,"drinkIds":[1]}'

# 8) Verify
curl -k https://localhost:7047/api/selection/active-drinks
